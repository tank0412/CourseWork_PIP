create table users
(
	id bigserial not null
		constraint users_pkey
			primary key,
	nickname varchar(512),
	name varchar(512),
	description text,
	registrationdate timestamp with time zone default now(),
	password varchar(30),
	hash bigint default 0,
	jid varchar(256),
	isadmin boolean
)
;


create table rent
(
	id bigserial not null
		constraint rent_pkey
			primary key,
	car_id bigserial not null
		constraint "rent_Car_ID_fkey"
			references cars,
	client_id bigserial not null
		constraint "rent_Client_ID_fkey"
			references users,
	date_start_rent timestamp with time zone default now() not null,
	date_end_rent timestamp with time zone default now() not null,
	price_of_rent bigint not null,
	isprolongated boolean,
	isconfirmed boolean
)
;

create table fines
(
	id bigserial not null
		constraint fines_pkey
			primary key,
	"ñlient_id" bigserial not null
		constraint "fines_Client_ID_fkey"
			references users,
	date_get_fine timestamp with time zone default now() not null,
	reason_of_fine varchar(40) not null,
	fine_price bigint not null
)
;



create table cars
(
	id bigserial not null
		constraint cars_pkey
			primary key
)
;
